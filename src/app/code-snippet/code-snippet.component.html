    export class AppComponent {

        constructor(
            private iconRegistry: MatIconRegistry,
            private sanitizer: DomSanitizer
        ) {

            this.iconRegistry.addSvgIcon(
                'my-star',
                this.sanitizer.bypassSecurityTrustResourceUrl('assets/star.svg')
            );

        }
    }
